{{- /*
Post-install message for your Helm chart.
Prints how to access the application depending on the service type or ingress configuration.
*/ -}}

{{- if and (hasKey .Values "ingress") .Values.ingress.enabled }}
  {{- range $host := .Values.ingress.hosts }}
    {{- range .paths }}
      echo "Visit http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }} to use your application"
    {{- end }}
  {{- end }}

{{- else if and (hasKey .Values "service") (contains "NodePort" .Values.service.type) }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} \
    -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "express-rest-demo-helm.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} \
    -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Visit http://$NODE_IP:$NODE_PORT to use your application"

{{- else if and (hasKey .Values "service") (contains "LoadBalancer" .Values.service.type) }}
  echo "NOTE: It may take a few minutes for the LoadBalancer IP to be available."
  echo "      You can watch its status by running:"
  echo "      kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "express-rest-demo-helm.fullname" . }}"
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} \
    {{ include "express-rest-demo-helm.fullname" . }} \
    --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Visit http://$SERVICE_IP:{{ .Values.service.port }} to use your application"

{{- else if and (hasKey .Values "service") (contains "ClusterIP" .Values.service.type) }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} \
    -l "app.kubernetes.io/name={{ include "express-rest-demo-helm.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" \
    -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME \
    -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  echo "Forwarding port..."
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT

{{- else }}
  echo "⚠️ No valid exposure method found (Ingress, NodePort, LoadBalancer, or ClusterIP)."
  echo "Please check your values.yaml configuration."
{{- end }}
